-- Função Auxiliar para Cantos Arredondados
local function AddUICorner(parent, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius or 6)
    corner.Parent = parent
    return corner
end

-- Função Auxiliar para Aplicar o Gradiente nos Textos
local function ApplyTextGradient(textObject)
    local gradient = Instance.new("UIGradient")
    gradient.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, Color3.fromRGB(100, 200, 255)), -- Azul Claro
        ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 50, 50))    -- Vermelho
    })
    gradient.Parent = textObject
    textObject.TextColor3 = Color3.fromRGB(255, 255, 255)
end

-- Função União de Arrasto Suave (Resolve problemas de sobreposição de botões)
local function MakeDraggable(dragHandle, objectToMove)
    local dragging = false
    local dragInput, dragStart, startPos

    dragHandle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = objectToMove.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    dragHandle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseBehavior or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    RunService.RenderStepped:Connect(function()
        if dragging and dragInput then
            local delta = dragInput.Position - dragStart
            objectToMove.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

        StatusLabel.Text = count .. " jogador(es) na lista" 
    end
end

-- Mecânicas de Busca das Ferramentas
local function getWeapon()
    local char = Player.Character
    if char and char:FindFirstChild("Kawaii Revolver") then return char:FindFirstChild("Kawaii Revolver") end
    if Player.Backpack:FindFirstChild("Kawaii Revolver") then return Player.Backpack:FindFirstChild("Kawaii Revolver") end
    return nil
end

local function isHoldingWeapon()
    local char = Player.Character
    if char and char:FindFirstChild("Kawaii Revolver") then
        if not canShoot and tick() - lastShot < 0.05 then return false end
        return true
    end
    return false
end

local function getPencilWeapon()
    local char = Player.Character
    if char and char:FindFirstChild("Pencil") then return char:FindFirstChild("Pencil") end
    if Player.Backpack:FindFirstChild("Pencil") then return Player.Backpack:FindFirstChild("Pencil") end
    return nil
end

local function isHoldingPencil()
    local char = Player.Character
    if char and char:FindFirstChild("Pencil") then
        if not canShootPencil and tick() - lastPencilShot < 0.05 then return false end
        return true
    end
    return false
end

local function verificarAlvoValido(playerName)
    local targetPlayer = Players:FindFirstChild(playerName)
    if not targetPlayer then return nil, nil end
    local char = targetPlayer.Character
    if not char then return nil, nil end
    if char:FindFirstChildOfClass("ForceField") then return nil, nil end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum and hum.Health > 0 then
        return targetPlayer, hum
    end
    return nil, nil
end

local function ForcarBotaoPencil()
    local function getButton()
        local playerGui = Player:WaitForChild("PlayerGui", 2)
        if not playerGui then return nil end
        local menuScreen = playerGui:FindFirstChild("Menu Screen")
        local frame = menuScreen and menuScreen:FindFirstChild("Frame")
        local scrollingFrame = frame and frame:FindFirstChild("ScrollingFrame")
        return scrollingFrame and scrollingFrame:FindFirstChild("Pencil")
    end
    local function makeVisible(object)
        while object and object:IsA("GuiObject") do
            object.Visible = true
            object = object.Parent
        end
    end
    local btn = getButton()
    if btn and btn.Parent then
        makeVisible(btn)
        if getconnections then
            for _, connection in pairs(getconnections(btn.MouseButton1Click)) do
                pcall(function() connection:Fire() end)
            end
        end
    end
end

-- Mecânica de Fling
local function ExecutarFling(TargetPlayer)
    local Character = Player.Character
    local Humanoid = Character and Character:FindFirstChildOfClass("Humanoid")
    local RootPart = Humanoid and Humanoid.RootPart
    local TCharacter = TargetPlayer.Character
    if not TCharacter then return end
    
    local THumanoid, TRootPart, THead, Accessory, Handle
    if TCharacter:FindFirstChildOfClass("Humanoid") then THumanoid = TCharacter:FindFirstChildOfClass("Humanoid") end
    if THumanoid and THumanoid.RootPart then TRootPart = THumanoid.RootPart end
    if TCharacter:FindFirstChild("Head") then THead = TCharacter.Head end
    if TCharacter:FindFirstChildOfClass("Accessory") then Accessory = TCharacter:FindFirstChildOfClass("Accessory") end
    if Accessory and Accessory:FindFirstChild("Handle") then Handle = Accessory.Handle end
    
    if Character and Humanoid and RootPart then
        if RootPart.Velocity.Magnitude < 50 then getgenv().OldPos = RootPart.CFrame end
        if THumanoid and THumanoid.Sit then return end
        
        if THead then workspace.CurrentCamera.CameraSubject = THead
        elseif Handle then workspace.CurrentCamera.CameraSubject = Handle
        elseif THumanoid and TRootPart then workspace.CurrentCamera.CameraSubject = THumanoid end
        
        if not TCharacter:FindFirstChildWhichIsA("BasePart") then return end
        
        local FPos = function(BasePart, Pos, Ang)
            RootPart.CFrame = CFrame.new(BasePart.Position) * Pos * Ang
            Character:SetPrimaryPartCFrame(CFrame.new(BasePart.Position) * Pos * Ang)
            RootPart.Velocity = Vector3.new(9e7, 9e7 * 10, 9e7)
            RootPart.RotVelocity = Vector3.new(9e8, 9e8, 9e8)
        end
        
        local SFBasePart = function(BasePart)
            local TimeToWait = 1.5 
            local Time = tick()
            local Angle = 0
            repeat
                if RootPart and THumanoid then
                    if BasePart.Velocity.Magnitude < 50 then
                        Angle = Angle + 100
                        FPos(BasePart, CFrame.new(0, 1.5, 0) + THumanoid.MoveDirection * BasePart.Velocity.Magnitude / 1.25, CFrame.Angles(math.rad(Angle),0 ,0))
                        task.wait()
                        FPos(BasePart, CFrame.new(0, -1.5, 0) + THumanoid.MoveDirection * BasePart.Velocity.Magnitude / 1.25, CFrame.Angles(math.rad(Angle), 0, 0))
                        task.wait()
                    else
                        FPos(BasePart, CFrame.new(0, 1.5, THumanoid.WalkSpeed), CFrame.Angles(math.rad(90), 0, 0))
                        task.wait()
                        FPos(BasePart, CFrame.new(0, -1.5, -THumanoid.WalkSpeed), CFrame.Angles(0, 0, 0))
                        task.wait()
                    end
                end
            until Time + TimeToWait < tick() or not loopActive
        end
        
        workspace.FallenPartsDestroyHeight = 0/0
        
        local BV = Instance.new("BodyVelocity")
        BV.Parent = RootPart
        BV.Velocity = Vector3.new(0, 0, 0)
        BV.MaxForce = Vector3.new(9e9, 9e9, 9e9)
        
        Humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
        
        if TRootPart then SFBasePart(TRootPart)
        elseif THead then SFBasePart(THead)
        elseif Handle then SFBasePart(Handle) end
        
        BV:Destroy()
        Humanoid:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
        workspace.CurrentCamera.CameraSubject = Humanoid
        
        if getgenv().OldPos then
            repeat
                RootPart.CFrame = getgenv().OldPos * CFrame.new(0, .5, 0)
                Character:SetPrimaryPartCFrame(getgenv().OldPos * CFrame.new(0, .5, 0))
                Humanoid:ChangeState("GettingUp")
                for _, part in pairs(Character:GetChildren()) do
                    if part:IsA("BasePart") then part.Velocity, part.RotVelocity = Vector3.new(), Vector3.new() end
                end
                task.wait()
            until (RootPart.Position - getgenv().OldPos.Position).Magnitude < 25
            workspace.FallenPartsDestroyHeight = getgenv().FPDH
        end
    end
end

-- Loops de Execução em Segundo Plano
task.spawn(function()
    while true do
        task.wait(0.1)
        if loopActive then
            for playerName, _ in pairs(SelectedTargets) do
                if not loopActive then break end
                local target, _ = verificarAlvoValido(playerName)
                if target then ExecutarFling(target) end
            end
        end
    end
end)

RunService.Heartbeat:Connect(function()
    if not loopActive or not canShoot then return end
    local char = Player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    
    local weapon = getWeapon()
    local agora = tick()
    
    if not weapon then
        ForcarBotaoPencil()
        lastShot = agora
        return 
    end
    
    local cooldownAtual = cooldown
    if isHoldingWeapon() then cooldownAtual = cooldown + 1.0 end
    
    if agora - lastShot >= cooldownAtual then
        for playerName, _ in pairs(SelectedTargets) do
            local target, hum = verificarAlvoValido(playerName)
            if target and hum then
                local remote = weapon:FindFirstChild("DamageRemote")
                if remote then
                    canShoot = false
                    lastShot = agora
                    pcall(function()
                        weapon.Parent = char
                        remote:FireServer(hum)
                        weapon.Parent = Player.Backpack
                    end)
                    canShoot = true
                    break 
                end
            end
        end
    end
end)

RunService.Heartbeat:Connect(function()
    if not loopActive or not canShootPencil then return end
    local char = Player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end
    
    local pencilWeapon = getPencilWeapon()
    local agora = tick()
    
    if not pencilWeapon then
        ForcarBotaoPencil()
        lastPencilShot = agora
        return 
    end
    
    local cooldownAtual = pencilCooldown
    if isHoldingPencil() then cooldownAtual = pencilCooldown + 0.1 end
    
    if agora - lastPencilShot >= cooldownAtual then
        for playerName, _ in pairs(SelectedTargets) do
            local target, hum = verificarAlvoValido(playerName)
            if target and hum then
                local remote = pencilWeapon:FindFirstChild("DamageRemote") or pencilWeapon:FindFirstChildWhichIsA("RemoteEvent")
                if remote then
                    canShootPencil = false
                    lastPencilShot = agora
                    pcall(function()
                        pencilWeapon.Parent = char
                        remote:FireServer(hum)
                        pencilWeapon.Parent = Player.Backpack
                    end)
                    canShootPencil = true
                    break 
                end
            end
        end
    end
end)
